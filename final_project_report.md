class FirebaseService {
  final FirebaseFirestore _firestore = FirebaseFirestore.instance;

  Future<List<Product>> getProducts({String? category}) async {
    Query query = _firestore.collection('products');

    if (category != null) {
      query = query.where('category', isEqualTo: category);
    }

    final snapshot = await query.get();
    return snapshot.docs.map((doc) => Product.fromFirestore(doc)).toList();
  }

  Future<void> createOrder(Order order) async {
    await _firestore.collection('orders').doc(order.id).set(order.toFirestore());
  }
}

## APPENDIX C: FIREBASE SECURITY RULES

### Firestore Security Rules
```
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // User profiles - only authenticated users can read/write their own profile
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Customer data - shop owners and employees can manage
    match /customers/{customerId} {
      allow read, write: if isShopOwner() || isEmployee();
    }

    // Orders - customers can read their orders, shop staff can manage all
    match /orders/{orderId} {
      allow read: if request.auth != null &&
                   (request.auth.uid == resource.data.customerId ||
                    isShopOwner() || isEmployee());
      allow create: if request.auth != null;
      allow update, delete: if isShopOwner() || isEmployee();
    }

    // Products - public read, shop owner management
    match /products/{productId} {
      allow read: if true;
      allow write: if isShopOwner();
    }

    // Chat messages - authenticated users only
    match /chat/{conversationId}/messages/{messageId} {
      allow read, write: if request.auth != null;
    }
  }

  // Helper functions
  function isShopOwner() {
    return request.auth != null &&
           exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
           get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'shop_owner';
  }

  function isEmployee() {
    return request.auth != null &&
           exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
           get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'employee';
  }
}
```

           firestore.exists(/databases/(default)/documents/users/$(request.auth.uid)) &&
           firestore.get(/databases/(default)/documents/users/$(request.auth.uid)).data.role == 'employee';
  }
}
